```python
    def calculate_for_user(self, user, memory):
        base_rtc = memory.collective_rtc  # RTC universale
        user_adjustment = user.preferences.rtc_weights
        
        return blend(base_rtc, user_adjustment, ratio=0.7)
        # 70% collettivo, 30% personale
```
